{
The new algorithm calculates the heuristics matrix by taking the square root of the product of distances of the two nodes in the distance matrix.
}

import numpy as np

def heuristics(distance_matrix):
    num_nodes = distance_matrix.shape[0]
    heuristics_matrix = np.zeros((num_nodes, num_nodes))

    for i in range(num_nodes):
        for j in range(num_nodes):
            if i != j:
                heuristics_matrix[i, j] = np.sqrt(distance_matrix[i, i] * distance_matrix[j, j])

    return heuristics_matrix
