import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros((n, n))

    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            heuristics_matrix[i][j] = 1 / (distance_matrix[i][j] + 1)

    return heuristics_matrix
