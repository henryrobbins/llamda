```python
import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Combines mean distance and inverse distance to score edges in TSP, prioritizing shorter edges
    while maintaining simplicity and avoiding normalization complexities.
    """
    # Calculate the mean distance for each node
    mean_distances = np.mean(distance_matrix, axis=1)
    
    # Calculate the promising scores based on inverse distances
    with np.errstate(divide='ignore', invalid='ignore'):
        promising_scores = mean_distances[:, np.newaxis] / distance_matrix
    
    # Set unpromising distances (higher than mean) to zero
    promising_scores[distance_matrix > mean_distances[:, np.newaxis]] = 0
    
    return promising_scores
```
