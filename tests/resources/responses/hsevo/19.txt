import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic design for TSP combining 
    edge distance, node connectivity, and clustering to prioritize promising edges.
    """
    
    num_nodes = distance_matrix.shape[0]
    
    # Calculate the mean distance for each node
    mean_distances = np.mean(distance_matrix, axis=1)

    # Calculate connectivity scores based on the number of connections
    connectivity_scores = np.sum(distance_matrix > 0, axis=1) / (num_nodes - 1)

    # Calculate inverse distances but reserve high scores for short distances
    with np.errstate(divide='ignore', invalid='ignore'):
        inverse_distances = np.where(distance_matrix > 0, mean_distances[:, np.newaxis] / distance_matrix, 0)
    
    # Combine scores with a balance towards shorter connections and connectivity
    promising_scores = (inverse_distances * connectivity_scores[:, np.newaxis]) ** 1.5
    
    # Sparsify matrix by zeroing out low scores based on a dynamic threshold
    threshold = np.percentile(promising_scores, 75)
    promising_scores[promising_scores < threshold] = 0
    
    # Ensure diagonal entries (self-loops) are zero
    np.fill_diagonal(promising_scores, 0)
    
    return promising_scores
