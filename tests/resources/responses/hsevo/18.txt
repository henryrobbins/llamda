import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic design for TSP using a combination of 
    relative distances, node centrality, and edge density to enhance 
    edge selection for promising paths more effectively.
    """
    
    # Calculate the mean distance for each node
    mean_distances = np.mean(distance_matrix, axis=1)

    # Calculate the variance of distances to understand spread
    distance_variance = np.var(distance_matrix, axis=1)

    # Calculate metrics that balance proximity and node centrality
    proximity_scores = mean_distances / (distance_variance + 1e-5)  # Avoid division by zero
    
    # Calculate inverse distances while avoiding infinity values
    with np.errstate(divide='ignore', invalid='ignore'):
        inverse_distances = np.where(distance_matrix > 0, 1 / distance_matrix, 0)

    # Combine the scores to emphasize promising edges
    promising_scores = proximity_scores[:, np.newaxis] * inverse_distances
    
    # Sparsify the matrix by zeroing out unpromising scores (threshold can be adjusted for application)
    promising_scores[promising_scores < 0.05] = 0
    
    # Ensure diagonal entries (self-loops) are zero
    np.fill_diagonal(promising_scores, 0)
    
    return promising_scores
