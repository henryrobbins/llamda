import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Calculate promising scores for edges in TSP by combining inverse distances
    and mean-based normalization, while setting low scores to zero.
    """
    # Calculate the inverse of the distance matrix
    inverse_distances = 1 / distance_matrix
    
    # Calculate the mean distances for each node
    mean_distances = np.mean(distance_matrix, axis=1)
    
    # Combine the two heuristics
    promising_scores = inverse_distances * (mean_distances[:, np.newaxis] / mean_distances.max())
    
    # Set unpromising edges to zero
    promising_scores[np.isnan(promising_scores) | (promising_scores < 0.1)] = 0
    
    return promising_scores
