```python
import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Heuristic function to estimate edge importance by combining inverses of distances
    while sparsifying low-value connections to streamline decision-making for TSP.
    """
    # Small value to avoid division by zero in edge cases
    eps = 1e-6
    
    # Compute inverted distances
    inverted_distances = 1 / (distance_matrix + eps)
    
    # Set a threshold to sparsify the matrix: retains top 50% edges
    threshold = np.percentile(inverted_distances, 50, axis=1)
    
    # Create heuristics matrix with sparsifying condition
    heuristics_matrix = np.where(inverted_distances >= threshold[:, np.newaxis], inverted_distances, 0)
    
    return heuristics_matrix
```
