```python
import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Combines inverse distances and node mean distances to prioritize edges, avoiding complexity.
    Outliers are reduced by sparsing unpromising edges, enhancing selection clarity.
    """
    eps = 1e-6  # Small value to avoid division by zero
    # Calculate the inverse distances
    inverted_distances = 1 / (distance_matrix + eps)
    
    # Calculate mean distances for each node
    mean_distances = np.mean(distance_matrix, axis=1)
    
    # Create promising scores by combining the two metrics
    promising_scores = mean_distances[:, np.newaxis] * inverted_distances
    
    # Sparsify the matrix by setting unpromising elements to zero
    promising_scores[promising_scores < np.mean(promising_scores)] = 0
    
    return promising_scores
```
