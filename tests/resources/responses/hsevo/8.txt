import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Calculate edge heuristics for TSP by combining inverse distances
    and row normalization, while setting unpromising elements to zero.
    """
    eps = 1e-6  # Small value to avoid division by zero
    inverted_distances = 1 / (distance_matrix + eps)
    
    # Use the inverse distances directly but set values to zero for longer edges
    inverted_distances[distance_matrix > np.percentile(distance_matrix, 75)] = 0
    
    # Normalize rows to retain relative Prominence
    row_sum = inverted_distances.sum(axis=1)
    heuristics_matrix = inverted_distances / (row_sum[:, np.newaxis] + eps)
    
    return heuristics_matrix
