import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic design for edge selection based on relative contributions
    of distances and node connectivity, enhancing the identification of promising edges
    in the context of the Traveling Salesman Problem (TSP).
    """
    
    # Calculate the mean distances for each node
    mean_distances = np.mean(distance_matrix, axis=1)
    
    # Influence of each edge determined by how much closer it gets to the mean distance
    closeness_factor = mean_distances[:, np.newaxis] - distance_matrix
    
    # Exclude edges that do not connect two different nodes
    closeness_factor[distance_matrix == 0] = -np.inf
    
    # Compute a score based on the closeness factor, emphasizing connections that approach mean distances
    promising_scores = np.clip(closeness_factor, 0, None) ** 2
    
    # Combine with an inverse distance to further emphasize shorter, more beneficial edges
    with np.errstate(divide='ignore', invalid='ignore'):
        inverse_distances = np.where(distance_matrix > 0, 1 / distance_matrix, 0)

    # Calculate the final promising scores by combining both approaches (weighted)
    final_scores = promising_scores * inverse_distances

    # Apply a sparsification process to zero-out unpromising scores
    threshold = np.percentile(final_scores, 75)  # Keep only the top 25% of edge scores
    final_scores[final_scores < threshold] = 0
    
    # Ensure diagonal entries (self-loops) are zero
    np.fill_diagonal(final_scores, 0)
    
    return final_scores
