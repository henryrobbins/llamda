import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic design prioritizing edge selection
    based on relative distance efficiency and node connectivity.
    """
    
    # Calculate the mean distance for each node
    mean_distances = np.mean(distance_matrix, axis=1)
    
    # Calculate the relative importance of each edge based on distance
    # using the mean distance to normalize scores
    relative_scores = mean_distances[:, np.newaxis] / (distance_matrix + 1e-5)  # Avoid division by zero
    
    # Calculate connectivity score based on the number of edges connecting to each node
    connectivity_scores = np.sum(distance_matrix > 0, axis=1)[:, np.newaxis]
    
    # Combine the relative scores with the connectivity scores
    combined_scores = relative_scores * connectivity_scores
    
    # Apply a non-linear transformation for emphasis on selecting shorter, more connected edges
    promising_scores = combined_scores ** 1.5 
    
    # Sparsify matrix by zeroing out unpromising scores (threshold can be adjusted)
    promising_scores[promising_scores < 0.2] = 0
    
    # Ensure diagonal entries (self-loops) are zero
    np.fill_diagonal(promising_scores, 0)
    
    return promising_scores
