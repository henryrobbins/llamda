import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    # Calculate the mean distance for each node
    mean_distances = np.mean(distance_matrix, axis=1)
    # Calculate the inverse of distances weighted by mean distances
    with np.errstate(divide='ignore', invalid='ignore'):
        promising_scores = mean_distances[:, np.newaxis] / distance_matrix
        promising_scores[np.isnan(promising_scores)] = 0  # Handle infinity values
    return promising_scores
