import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    eps = 1e-6  # Small value to avoid division by zero
    inverted_distances = 1 / (distance_matrix + eps)
    row_sum = inverted_distances.sum(axis=1)
    heuristics_matrix = inverted_distances / row_sum[:, np.newaxis]
    return heuristics_matrix
