import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    # Applying a modified version of the original heuristic
    with np.errstate(divide='ignore', invalid='ignore'):
        filter_nan = np.where(distance_matrix == 0, np.nan, distance_matrix)  # to avoid division by zero
        heuristic_scores = np.nan_to_num(1 / filter_nan, nan=0)  # Replace NaN with 0
    # Modifying with additional heuristic criteria: prefer shorter edges with more connections
    connectivity = np.reciprocals(np.sum(distance_matrix > 0, axis=0), where=(np.sum(distance_matrix > 0, axis=0) > 0))
    return heuristic_scores * connectivity
