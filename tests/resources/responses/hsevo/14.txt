import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Combine inverse of distances and mean distance factors for promising edge selection.
    Simplifies decision making while maintaining the core relationships in data.
    """

    # Calculate the mean distance for each node
    mean_distances = np.mean(distance_matrix, axis=1)
    
    # Calculate the inverse of distances
    with np.errstate(divide='ignore', invalid='ignore'):
        inverse_distances = mean_distances[:, np.newaxis] / distance_matrix
        inverse_distances[np.isnan(inverse_distances)] = 0  # Handle infinity values
    
    # Combine the inverse distances with a sparse threshold
    promising_scores = inverse_distances * (distance_matrix < np.mean(distance_matrix, axis=1, keepdims=True))

    # Ensure diagonal (self-loops) entries are zero
    np.fill_diagonal(promising_scores, 0)

    return promising_scores
