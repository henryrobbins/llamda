```python
import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Calculate promising scores for edges in TSP.
    Combines inverse distances and mean distances for a clearer evaluation of edge potential.
    """
    # Calculate inverse distances
    inverse_distances = 1 / distance_matrix
    
    # Calculate mean distances
    mean_distances = np.mean(distance_matrix, axis=1)
    
    # Calculate promising scores considering both factors
    promising_scores = inverse_distances * mean_distances[:, np.newaxis]
    
    # Sparsify the matrix by setting unpromising elements to zero
    promising_scores[distance_matrix == 0] = 0  # Remove self-loops
    promising_scores[promising_scores < 0.1] = 0  # Set threshold
    
    return promising_scores
```
