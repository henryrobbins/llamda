{
The new algorithm will prioritize selecting the next node based on both the shortest distance and the highest heuristics value in the row of the current node in the heuristics matrix, aiming to balance exploration and exploitation in finding a solution to the Traveling Salesman Problem.
}

```python
import numpy as np

def heuristics(distance_matrix):
    n = len(distance_matrix)
    heuristics_matrix = np.zeros((n, n))
    
    for _ in range(n):
        available_nodes = list(range(n))
        current_node = np.random.choice(available_nodes)
        available_nodes.remove(current_node)
        
        while available_nodes:
            distances = [distance_matrix[current_node][x] for x in available_nodes]
            heuristics_row = heuristics_matrix[current_node]
            heuristic_scores = [1 / (1 + heuristics_row[x]) + 1 / (1 + distances[i]) for i, x in enumerate(available_nodes)]
            next_node = available_nodes[np.argmax(heuristic_scores)]
            heuristics_matrix[current_node][next_node] += 1
            heuristics_matrix[next_node][current_node] += 1
            available_nodes.remove(next_node)
            current_node = next_node
    
    return heuristics_matrix
```
