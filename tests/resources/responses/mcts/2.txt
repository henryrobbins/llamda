{
The new algorithm selects edges randomly based on a probability distribution calculated from the distances between nodes, iteratively updating the indicators to create a stochastic sampling approach for solving the Traveling Salesman Problem.
}

import numpy as np

def heuristics(distance_matrix):
    n = len(distance_matrix)
    heuristics_matrix = np.zeros((n, n))
    probabilities = np.zeros((n, n))

    for i in range(n):
        for j in range(n):
            if i != j:
                probabilities[i][j] = 1 / distance_matrix[i][j]
    
    for _ in range(n):
        available_nodes = list(range(n))
        current_node = np.random.choice(available_nodes)
        available_nodes.remove(current_node)
        
        while available_nodes:
            next_node = np.random.choice(available_nodes, p=probabilities[current_node]/np.sum(probabilities[current_node]))
            heuristics_matrix[current_node][next_node] += 1
            heuristics_matrix[next_node][current_node] += 1
            available_nodes.remove(next_node)
            current_node = next_node
    
    return heuristics_matrix
