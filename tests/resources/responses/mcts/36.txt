{Design Idea: The new algorithm will prioritize edge selection based on the ratio of distances to the number of times an edge has been included in the solution, given a starting node.}

```python
import numpy as np

def heuristics(distance_matrix):
    n = len(distance_matrix)
    heuristics_matrix = np.zeros((n, n))
    
    for _ in range(n):
        available_nodes = list(range(n))
        current_node = np.random.choice(available_nodes)
        available_nodes.remove(current_node)
        
        while available_nodes:
            distances = [distance_matrix[current_node][x] for x in available_nodes]
            counts = [heuristics_matrix[current_node][x] + heuristics_matrix[x][current_node] for x in available_nodes]
            scores = [distances[i] / counts[i] if counts[i] > 0 else distances[i] for i in range(len(available_nodes))]
            next_node = available_nodes[scores.index(min(scores))]
            
            heuristics_matrix[current_node][next_node] += 1
            heuristics_matrix[next_node][current_node] += 1
            available_nodes.remove(next_node)
            current_node = next_node
    
    return heuristics_matrix
```
